% to keep this somewhat sane
\newcommand{\homebuttons}{
    % https://tex.stackexchange.com/questions/187993/how-to-position-beamer-button-exactly
    \begin{tikzpicture}[remember picture, overlay]
        \node[shift={(-0.5cm, -0.5cm)}]() at (current page.north east){
            \hyperlink{Usage}{\beamerbutton{Help}}
        };
        \node[shift={(-1.0cm, 0.5cm)}]() at (current page.south east){
            \hyperlink{AHome}{\beamerbutton{A}}
        };
        \node[shift={(-0.5cm, 0.5cm)}]() at (current page.south east){
            \hyperlink{BHome}{\beamerbutton{B}}
        };
    \end{tikzpicture}
}
\newcommand{\usebackground}[1]{
    % https://tex.stackexchange.com/questions/7916/how-to-insert-a-background-image-in-a-beamer-frame
    % https://tex.stackexchange.com/questions/475375/absolute-positioning-of-images-in-beamers-background-canvas?noredirect=1&lq=1
    \usebackgroundtemplate{
        \vbox to \paperheight{\vfil\hbox to \paperwidth{\hfil
            \tikz\node[opacity=1.0, inner sep=0pt]{\includegraphics[width=\paperwidth, height=\paperheight, keepaspectratio]{#1}};
        \hfil}\vfil}
    }
}
